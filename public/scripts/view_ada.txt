SELECT
    ada.id AS id,
    ada.ada_no AS ada_no,
    ada.ada_date AS ada_date,
    ada.fund_id AS fund_id,
    f.fund AS fund,
    fund_cluster,
    ada.bank_account_id AS bank_account_id,
    banks.bank_account_no AS bank_account_no,
    ada.check_no AS check_no,
    ada.total_ps_amount AS total_ps_amount,
    ada.total_mooe_amount AS total_mooe_amount,
    ada.total_co_amount AS total_co_amount,
    (total_ps_amount+total_mooe_amount+total_co_amount) as total_ada_amount,
    signatory1,signatory1_position,
    signatory2,signatory2_position,
    signatory3,signatory3_position,
    signatory4,signatory4_position,
    signatory5,signatory5_position,
    ada.date_transferred AS date_transferred,
    ada.is_active AS is_active,
    ada.is_deleted AS is_deleted,
    ada.created_at AS created_at,
    ada.updated_at AS updated_at
FROM ada
LEFT JOIN commonlibrariesdb.funds f ON ada.fund_id = f.id
LEFT JOIN library_bank_accounts banks ON  ada.bank_account_id = banks.id
