SELECT
    payees.id AS id,
    payees.parent_id AS parent_id,
    payees.payee_type_id AS payee_type_id,
    payeetype.payee_type AS payee_type,
    payees.payee AS payee,
    payees.organization_name AS organization_name,
    payees.organization_acronynm AS organization_acronynm,
    payees.last_name AS last_name,
    payees.first_name AS first_name,
    payees.middle_initial AS middle_initial,
    payees.tin AS tin,
    payees.bank_id AS bank_id,
    banks.bank AS bank,
    banks.bank_acronym AS bank_acronym,
    banks.short_name AS bank_short_name,
    payees.bank_branch AS bank_branch,
    payees.bank_account_name AS bank_account_name,
    payees.bank_account_name1 AS bank_account_name1,
    payees.bank_account_name2 AS bank_account_name2,
    payees.bank_account_no AS bank_account_no,
    payees.is_verified AS is_verified,
    payees.is_lbp_enrolled AS is_lbp_enrolled,
    payees.address AS address,
    payees.office_address AS office_address,
    payees.email_address AS email_address,
    payees.contact_no AS contact_no,
    for_remit,
    bg_color,
    payees.is_active AS is_active,
    payees.is_deleted AS is_deleted,
    payees.created_at AS created_at,
    payees.updated_at AS updated_at,
    deleted_at
FROM fmsdb.library_payees payees
LEFT JOIN fmsdb.library_payee_type payeetype ON payees.payee_type_id = payeetype.id
LEFT JOIN commonlibrariesdb.banks ON payees.bank_id = banks.id
