SELECT
    c.id AS id,
    c.check_date AS check_date,
    c.check_no AS check_no,
    dv1.rs_no AS rs_no,
    c.dv_id AS dv_id,
    dv.dv_no AS dv_no,
    dv.division_acronym AS dv_division_acronym,
    dv.payee AS payee,
    dv.total_dv_net_amount AS total_dv_net_amount,
    c.fund_id AS fund_id,
    f.fund AS fund,
    c.bank_account_id AS bank_account_id,
    b.bank_account_no AS bank_account_no,
    bank_acronym,
    fund_cluster,
    c.is_active AS is_active,
    c.is_deleted AS is_deleted
FROM view_check_dv c
LEFT JOIN view_dvs dv ON c.dv_id = dv.id
LEFT JOIN view_dv_rs_net dv1 ON dv.id = dv1.dv_id
LEFT JOIN commonlibrariesdb.funds f ON dv.fund_id = f.id
LEFT JOIN view_library_bank_accounts b ON c.bank_account_id = b.id
WHERE c.dv_id IS NOT NULL GROUP BY dv.id
